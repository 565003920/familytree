# 族谱网站 - 完整数据库设计

## 一、用户与权限模块

### 1.1 用户表 (users)
```sql
CREATE TABLE users (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  username VARCHAR(50) UNIQUE NOT NULL COMMENT '用户名',
  phone VARCHAR(20) UNIQUE COMMENT '手机号',
  email VARCHAR(100) UNIQUE COMMENT '邮箱',
  password_hash VARCHAR(255) COMMENT '密码哈希',
  avatar_url VARCHAR(500) COMMENT '头像URL',
  nickname VARCHAR(50) COMMENT '昵称',
  real_name VARCHAR(50) COMMENT '真实姓名',
  gender ENUM('male', 'female', 'unknown') DEFAULT 'unknown',
  birth_date DATE COMMENT '出生日期',
  status TINYINT DEFAULT 1 COMMENT '状态：1-正常 0-禁用',
  last_login_at TIMESTAMP COMMENT '最后登录时间',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  INDEX idx_phone (phone),
  INDEX idx_email (email),
  INDEX idx_username (username)
) COMMENT='用户表';
```

### 1.2 用户角色表 (user_roles)
```sql
CREATE TABLE user_roles (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  user_id BIGINT NOT NULL,
  family_id BIGINT NOT NULL,
  role_type ENUM('admin', 'branch_admin', 'member', 'guest') DEFAULT 'member',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES users(id),
  FOREIGN KEY (family_id) REFERENCES families(id),
  UNIQUE KEY uk_user_family (user_id, family_id)
) COMMENT='用户角色表';
```

### 1.3 权限表 (permissions)
```sql
CREATE TABLE permissions (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  role_type VARCHAR(50) NOT NULL,
  resource VARCHAR(100) NOT NULL COMMENT '资源：family_tree, album, activity等',
  action VARCHAR(50) NOT NULL COMMENT '操作：create, read, update, delete',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  UNIQUE KEY uk_role_resource_action (role_type, resource, action)
) COMMENT='权限表';
```


## 二、家族管理模块

### 2.1 家族表 (families)
```sql
CREATE TABLE families (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  name VARCHAR(100) NOT NULL COMMENT '家族名称',
  surname VARCHAR(50) NOT NULL COMMENT '姓氏',
  origin_place VARCHAR(200) COMMENT '祖籍地',
  founder_name VARCHAR(50) COMMENT '始祖姓名',
  founder_id BIGINT COMMENT '始祖成员ID',
  admin_id BIGINT NOT NULL COMMENT '族长用户ID',
  description TEXT COMMENT '家族简介',
  member_count INT DEFAULT 0 COMMENT '成员数量',
  generation_count INT DEFAULT 0 COMMENT '世代数',
  cover_image VARCHAR(500) COMMENT '封面图',
  status TINYINT DEFAULT 1 COMMENT '状态：1-正常 0-停用',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (admin_id) REFERENCES users(id),
  INDEX idx_surname (surname),
  INDEX idx_admin (admin_id)
) COMMENT='家族表';
```

### 2.2 家族成员表 (family_members)
```sql
CREATE TABLE family_members (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  family_id BIGINT NOT NULL,
  user_id BIGINT COMMENT '关联用户ID（可为空）',
  name VARCHAR(50) NOT NULL COMMENT '姓名',
  former_names VARCHAR(200) COMMENT '曾用名，多个用逗号分隔',
  alias VARCHAR(50) COMMENT '字号',
  gender ENUM('male', 'female', 'unknown') DEFAULT 'unknown',
  birth_date DATE COMMENT '出生日期',
  birth_place VARCHAR(200) COMMENT '出生地',
  death_date DATE COMMENT '逝世日期',
  death_place VARCHAR(200) COMMENT '逝世地',
  is_alive TINYINT DEFAULT 1 COMMENT '是否在世：1-是 0-否',
  generation VARCHAR(20) COMMENT '字辈',
  generation_order INT COMMENT '世代序号（第几代）',
  ranking INT COMMENT '排行（第几子/女）',
  education VARCHAR(100) COMMENT '学历',
  occupation VARCHAR(100) COMMENT '职业',
  title VARCHAR(100) COMMENT '职务',
  achievements TEXT COMMENT '成就荣誉',
  bio TEXT COMMENT '生平事迹',
  hobbies VARCHAR(500) COMMENT '兴趣爱好',
  religion VARCHAR(50) COMMENT '宗教信仰',
  political_status VARCHAR(50) COMMENT '政治面貌',
  avatar_url VARCHAR(500) COMMENT '头像',
  created_by BIGINT COMMENT '创建人',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  FOREIGN KEY (family_id) REFERENCES families(id),
  FOREIGN KEY (user_id) REFERENCES users(id),
  INDEX idx_family (family_id),
  INDEX idx_name (name),
  INDEX idx_generation (generation_order)
) COMMENT='家族成员表';
```


### 2.3 家族关系表 (family_relations)
```sql
CREATE TABLE family_relations (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  family_id BIGINT NOT NULL,
  member_id BIGINT NOT NULL COMMENT '成员A',
  related_member_id BIGINT NOT NULL COMMENT '成员B',
  relation_type ENUM('parent', 'spouse', 'child', 'sibling') NOT NULL,
  relation_desc VARCHAR(50) COMMENT '关系描述：父亲、母亲、配偶、长子等',
  start_date DATE COMMENT '关系开始日期（如结婚日期）',
  end_date DATE COMMENT '关系结束日期（如离婚日期）',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (family_id) REFERENCES families(id),
  FOREIGN KEY (member_id) REFERENCES family_members(id),
  FOREIGN KEY (related_member_id) REFERENCES family_members(id),
  INDEX idx_member (member_id),
  INDEX idx_related (related_member_id),
  INDEX idx_type (relation_type)
) COMMENT='家族关系表';
```

### 2.4 家族分支表 (family_branches)
```sql
CREATE TABLE family_branches (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  family_id BIGINT NOT NULL,
  branch_name VARCHAR(100) NOT NULL COMMENT '分支名称：长房、二房等',
  founder_member_id BIGINT COMMENT '分支始祖',
  parent_branch_id BIGINT COMMENT '父分支ID',
  branch_order INT COMMENT '排序',
  description TEXT,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (family_id) REFERENCES families(id),
  FOREIGN KEY (founder_member_id) REFERENCES family_members(id),
  INDEX idx_family (family_id)

```

```sql
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

  FOREIGN KEY (owner_id) REFERENCES users(id),
  INDEX idx_family (family_id),
  INDEX idx_owner (owner_id)

```

### 3.2 照片表 (photos)
```sql
CREATE TABLE photos (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
  album_id BIGINT NOT NULL,
  uploader_id BIGINT NOT NULL,
  title VARCHAR(200) COMMENT '照片标题',
  description TEXT COMMENT '照片描述',
  original_url VARCHAR(500) NOT NULL COMMENT '原图URL',
  thumbnail_url VARCHAR(500) COMMENT '缩略图URL',
  medium_url VARCHAR(500) COMMENT '中图URL',
  large_url VARCHAR(500) COMMENT '大图URL',
  file_size BIGINT COMMENT '文件大小（字节）',
  width INT COMMENT '宽度',
  height INT COMMENT '高度',
  format VARCHAR(20) COMMENT '格式：jpg, png等',
  taken_at TIMESTAMP COMMENT '拍摄时间',
  location VARCHAR(200) COMMENT '拍摄地点',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (album_id) REFERENCES albums(id),
  FOREIGN KEY (uploader_id) REFERENCES users(id),
  INDEX idx_album (album_id),
  INDEX idx_taken (taken_at)
) COMMENT='照片表';
```


### 3.3 照片标签表 (photo_tags)
```sql
CREATE TABLE photo_tags (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,

  tag_name VARCHAR(50) COMMENT '标签名称',

  FOREIGN KEY (member_id) REFERENCES family_members(id),
  INDEX idx_photo (photo_id),
  INDEX idx_member (member_id)

```

### 3.4 相册权限表 (album_permissions)
```sql
CREATE TABLE album_permissions (
  id BIGINT PRIMARY KEY AUTO_INCREMENT,
```
